Event 예제

데이터를 입력하면 바로 Event가 발생합니다..!!


	pragma solidity ^0.4.18;

	contract Coursetro {    
   		string fName;
   		uint age;
   
   		event Instructor(
       		string name,
       		uint age
    	);

   function setInstructor(string _fName, uint _age) public {
       fName = _fName;
       age = _age;
       Instructor(_fName, _age);        // Add this
   }
   
   function getInstructor() view public returns (string, uint) {
       return (fName, age);
   }
   
}
